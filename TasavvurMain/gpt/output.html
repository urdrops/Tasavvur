import pandas as pd
import matplotlib.pyplot as plt

# Загрузка данных из файла
data = pd.read_csv('file.csv')

# Получение значений столбца "ТЕРРИТОРИЯ, ОПИСАНИЕ УЧАСТКА"
territories = data['ТЕРРИТОРИЯ, ОПИСАНИЕ УЧАСТКА']

# Подсчет количества уникальных значений
territory_counts = territories.value_counts()

# Построение графика
plt.figure(figsize=(10, 6))
territory_counts.plot(kind='bar')
plt.title('Количество участков на каждой территории')
plt.xlabel('Территория')
plt.ylabel('Количество участков')
plt.xticks(rotation=45)
plt.show()
