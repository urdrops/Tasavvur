import pandas as pd
import matplotlib.pyplot as plt

#    
data = pd.read_csv('file.csv')

#    ",  "
territories = data[',  ']

#    
territory_counts = territories.value_counts()

#  
plt.figure(figsize=(10, 6))
territory_counts.plot(kind='bar')
plt.title('    ')
plt.xlabel('')
plt.ylabel(' ')
plt.xticks(rotation=45)
plt.show()
